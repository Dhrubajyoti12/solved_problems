#include<iostream>
#include<string>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int n=s.length();
    int a[n+1];
    a[0]=0;
    for(int i=1;i<=n;i++)
    {
        a[i]=a[i-1]^(1<<(s[i-1]-97));
       // cout<<a[i]<<" ";
    }
    int ans=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=i;j<=n;j++)
        {

            int p=a[i-1]^a[j];
            if((p&(p-1))==0)
                ans++;
        }

    }
    cout<<ans;
}
